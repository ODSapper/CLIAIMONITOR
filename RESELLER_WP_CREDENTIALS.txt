# View reseller profile
SELECT * FROM reseller_profiles WHERE id = 1;

# View all packages
SELECT id, name, slug, monthly_price, disk_quota_mb, bandwidth_quota_mb
FROM reseller_packages
WHERE reseller_id = 1;

# View all sites
SELECT
  ra.id,
  ra.username,
  ra.domain,
  pkg.name as package,
  u.email,
  ra.status
FROM reseller_accounts ra
JOIN users u ON u.id = ra.user_id
JOIN reseller_packages pkg ON pkg.id = ra.package_id
WHERE ra.reseller_id = 1
ORDER BY ra.id;

# View invoice details
SELECT
  i.invoice_number,
  i.total_amount,
  i.status,
  i.due_date,
  ii.description,
  ii.amount
FROM reseller_invoices i
JOIN reseller_invoice_items ii ON ii.reseller_invoice_id = i.id
WHERE i.reseller_id = 1;

# Complete reseller overview
SELECT
  rp.id as reseller_id,
  rp.company_name,
  rp.status as reseller_status,
  rp.tier,
  u.email as user_email,
  ra.username,
  ra.domain,
  pkg.name as package_name,
  pkg.monthly_price,
  ra.status as account_status
FROM reseller_profiles rp
LEFT JOIN reseller_accounts ra ON rp.id = ra.reseller_id
LEFT JOIN users u ON ra.user_id = u.id
LEFT JOIN reseller_packages pkg ON ra.package_id = pkg.id
ORDER BY ra.id;
